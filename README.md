## `// regras pessoais`

> essas não são dicas da internet. são aprendizados meus.

```python
class RegrasYslx0:

    def antes_de_escrever_codigo(self):
        """
        entenda o problema completamente.
        se não consegue explicar em uma frase, não entendeu ainda.
        """
        return self.entender_antes_de_codar()

    def durante_o_desenvolvimento(self):
        """
        commits pequenos e frequentes.
        nomes de variáveis que se explicam.
        funções que fazem uma coisa só.
        """
        regras = [
            "nunca commitar código quebrado na main",
            "nomear variáveis como se fosse ler daqui 6 meses",
            "uma função = uma responsabilidade. sempre.",
            "se precisou copiar e colar, hora de abstrair",
            "console.log é debug de dev, não de produção",
        ]
        return regras

    def quando_travar(self):
        """
        travou? levanta, bebe água, volta.
        rubber duck debugging funciona de verdade.
        """
        steps = [
            "1. leia o erro de verdade, palavra por palavra",
            "2. explique o problema em voz alta (sério)",
            "3. googla o erro exato com aspas",
            "4. docs oficial antes de stack overflow",
            "5. se ainda travado: pausa. sempre.",
        ]
        return steps

    def ao_revisar_codigo(self):
        """
        revise como se não fosse você que escreveu.
        seja honesto. seja cruel se precisar.
        """
        checklist = [
            "isso realmente precisa existir?",
            "tem efeito colateral escondido?",
            "está tratando todos os erros possíveis?",
            "vai escalar se triplicar o uso?",
            "alguém que não me conhece entende isso?",
        ]
        return checklist

    def ao_entregar(self):
        """
        entregue o que prometeu.
        se não consegue, comunica antes de atrasar.
        """
        return "feito > perfeito. mas feito com qualidade."
```

<div align="center">

### git — conventional commits

</div>

```bash
# formato que uso em todo projeto:

feat: adiciona autenticação JWT
fix: corrige bug na validação do formulário
docs: atualiza README com instruções de setup
refactor: extrai lógica de validação para service
style: formata código com prettier
test: adiciona testes para módulo de usuário
chore: atualiza dependências
perf: otimiza query do banco de dados

# estratégia de branches:
main          ← produção. nunca commita direto.
develop       ← integração de features
feat/nome     ← nova funcionalidade
fix/nome      ← correção de bug
hotfix/nome   ← emergência em produção
```

### código — como penso ao escrever

</div>

```typescript
// ❌ como não fazer:
const d = new Date();
const x = users.filter(u => u.a === true);
function calc(a: number, b: number) { return a * b * 0.1 + b; }

// ✅ como fazer:
const currentDate = new Date();
const activeUsers = users.filter(user => user.isActive === true);

const TAX_RATE = 0.1;
function calculateTotalWithTax(basePrice: number, quantity: number): number {
  const subtotal = basePrice * quantity;
  const tax = subtotal * TAX_RATE;
  return subtotal + tax;
}

// ❌ função que faz tudo:
function processUserAndSendEmailAndUpdateDB(user) { ... }

// ✅ responsabilidade única:
function validateUser(user: User): boolean { ... }
function formatUserEmail(user: User): EmailPayload { ... }
async function saveUser(user: User): Promise<void> { ... }
async function sendWelcomeEmail(email: EmailPayload): Promise<void> { ... }

// ❌ magic numbers:
if (user.age >= 18) { ... }
setTimeout(fn, 86400000);

// ✅ constantes com significado:
const LEGAL_AGE = 18;
const ONE_DAY_IN_MS = 24 * 60 * 60 * 1000;
if (user.age >= LEGAL_AGE) { ... }
setTimeout(fn, ONE_DAY_IN_MS);
```
